REM This script installs TightVNC Server 2.7.10
REM Server is the component which allows an endpoint to be viewed

IF EXIST "%PROGRAMFILES(X86)%" (GOTO 64BIT) ELSE (GOTO 32BIT)

:32BIT
msiexec /i \\10.70.2.21\eXpress\Scripts\VNC\tightvnc-2.7.10-setup-32bit.msi /quiet /norestart ADDLOCAL=Server SET_RUNCONTROLINTERFACE=1 VALUE_OF_RUNCONTROLINTERFACE=0 SET_QUERYACCEPTONTIMEOUT=1 VALUE_OF_QUERYACCEPTONTIMEOUT=1 SET_QUERYTIMEOUT=1 VALUE_OF_QUERYTIMEOUT=15 SET_VIEWONLYPASSWORD=1 SET_PASSWORD=1 VALUE_OF_PASSWORD=1234!@#$
echo.
echo.
echo     Installed TightVNC 32-bit

REM pause
GOTO END

:64BIT
msiexec /i \\10.70.2.21\eXpress\Scripts\VNC\tightvnc-2.7.10-setup-64bit.msi /quiet /norestart ADDLOCAL=Server SET_RUNCONTROLINTERFACE=1 VALUE_OF_RUNCONTROLINTERFACE=0 SET_QUERYACCEPTONTIMEOUT=1 VALUE_OF_QUERYACCEPTONTIMEOUT=1 SET_QUERYTIMEOUT=1 VALUE_OF_QUERYTIMEOUT=15 SET_VIEWONLYPASSWORD=1 SET_PASSWORD=1 VALUE_OF_PASSWORD=1234!@#$
echo.
echo.
echo     Installed TightVNC 64-bit

REM pause
GOTO END

:END
